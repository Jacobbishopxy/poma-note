\subsection*{Continuity and Compactness}

\begin{definition}
  A mapping $\mathbf{f}$ of a set $E$ into $R^k$ is said to be \textit{bounded} if there is a real number
  $M$ such that $|\mathbf{f}(x)| \le M$ for all $x \in E$.
\end{definition}
